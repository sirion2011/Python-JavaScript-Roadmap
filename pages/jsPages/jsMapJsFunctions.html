Что такое функции в JavaScript?

Функции — это блоки кода, предназначенные для многократного выполнения определённых задач. Они позволяют переиспользовать код, структурировать программы и создавать абстракции.

Объявление функции

function greet(name) {
        console.log("Привет, " + name);
      }
      
      greet("Анна"); // Привет, Анна

Функциональные выражения

Функцию можно присвоить переменной:

const sum = function(a, b) {
        return a + b;
      };
      
      console.log(sum(2, 3)); // 5

Стрелочные функции

const multiply = (a, b) => a * b;
      console.log(multiply(2, 3)); // 6

Параметры по умолчанию

function greet(name = "Гость") {
        console.log("Привет, " + name);
      }
      
      greet(); // Привет, Гость

Возврат значения

Оператор return завершает выполнение функции и возвращает значение:

function square(x) {
        return x * x;
      }
      
      console.log(square(4)); // 16

Функции обратного вызова (Callback)

function ask(question, yes, no) {
        if (confirm(question)) yes();
        else no();
      }
      
      ask("Вы согласны?", 
          () => alert("Вы согласились."),
          () => alert("Вы отменили выполнение."));

Замыкания

Замыкание — это функция, которая запоминает свою внешнюю среду, даже после завершения внешней функции:

function makeCounter() {
        let count = 0;
        return function() {
          return ++count;
        };
      }
      
      const counter = makeCounter();
      console.log(counter()); // 1
      console.log(counter()); // 2

IIFE (немедленно вызываемые функции)

(function() {
        console.log("Я вызываюсь сразу!");
      })();

Hoisting (всплытие)

Функции, объявленные через function, всплывают:

sayHi();
      
      function sayHi() {
        console.log("Привет");
      }

Rest-параметры

Позволяют передавать неопределённое количество аргументов:

function sumAll(...args) {
        return args.reduce((sum, val) => sum + val);
      }
      
      console.log(sumAll(1, 2, 3)); // 6

Деструктуризация параметров

function showUser({ name, age }) {
        console.log(`${name} (${age})`);
      }
      
      showUser({ name: "Анна", age: 22 }); // Анна (22)

Методы call, apply, bind

function sayHi() {
        console.log("Привет, " + this.name);
      }
      
      const user = { name: "Анна" };
      
      sayHi.call(user);  // Привет, Анна
      sayHi.apply(user); // Привет, Анна
      
      const bound = sayHi.bind(user);
      bound();           // Привет, Анна

Генераторы функций

function* generateSequence() {
        yield 1;
        yield 2;
        yield 3;
      }
      
      const generator = generateSequence();
      console.log(generator.next().value); // 1
      console.log(generator.next().value); // 2

Функции как замыкания для приватных переменных

function Counter() {
        let count = 0;
        return {
          increment() {
            count++;
            return count;
          },
          get() {
            return count;
          }
        };
      }
      
      const counter = Counter();
      console.log(counter.increment()); // 1
      console.log(counter.get());       // 1
